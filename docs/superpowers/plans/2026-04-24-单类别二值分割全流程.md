> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 为当前项目补齐单类别二值分割的大图分块推理主链路，稳定支持单张输入、目录批量、二值栅格输出和整图矢量化输出。

```cpp
#include "fusion/tile_planner.h"

#include <algorithm>
#include <stdexcept>

namespace gis_ai {

std::vector<TileWindow> TilePlanner::Plan(int raster_width, int raster_height, int tile_size, int overlap) const {
    if (raster_width <= 0 || raster_height <= 0 || tile_size <= 0) {
        throw std::invalid_argument("Invalid tile planning input");
    }
    if (overlap < 0 || overlap >= tile_size) {
        throw std::invalid_argument("Overlap must be in [0, tile_size)");
    }

    const int stride = tile_size - overlap;
    std::vector<int> xs;
    std::vector<int> ys;

    for (int x = 0; x < raster_width; x += stride) {
        xs.push_back(std::min(x, std::max(0, raster_width - tile_size)));
        if (xs.back() + tile_size >= raster_width) {
            break;
        }
    }

    for (int y = 0; y < raster_height; y += stride) {
        ys.push_back(std::min(y, std::max(0, raster_height - tile_size)));
        if (ys.back() + tile_size >= raster_height) {
            break;
        }
    }

    std::vector<TileWindow> windows;
    for (int y : ys) {
        for (int x : xs) {
            windows.push_back(TileWindow{
                x,
                y,
                std::min(tile_size, raster_width - x),
                std::min(tile_size, raster_height - y)
            });
        }
    }
    return windows;
}

} // namespace gis_ai
```

```cpp
std::unique_ptr<RasterData> RasterIO::ExtractWindow(const RasterData& data, const TileWindow& window) {
    auto result = std::make_unique<RasterData>();
    result->width = window.width;
    result->height = window.height;
    result->band_count = data.band_count;
    result->projection = data.projection;
    std::copy(std::begin(data.geotransform), std::end(data.geotransform), result->geotransform);
    result->geotransform[0] = data.geotransform[0] + window.x * data.geotransform[1] + window.y * data.geotransform[2];
    result->geotransform[3] = data.geotransform[3] + window.x * data.geotransform[4] + window.y * data.geotransform[5];
    result->bands.resize(data.band_count, std::vector<float>(window.width * window.height));

    for (int band = 0; band < data.band_count; ++band) {
        for (int y = 0; y < window.height; ++y) {
            for (int x = 0; x < window.width; ++x) {
                const int src_x = window.x + x;
                const int src_y = window.y + y;
                result->bands[band][y * window.width + x] = data.bands[band][src_y * data.width + src_x];
            }
        }
    }
    return result;
}
```

```cpp
#include "fusion/tile_merger.h"

#include <stdexcept>

namespace gis_ai {

TileMerger::TileMerger(int width, int height)
    : width_(width), height_(height), sums_(width * height, 0.0f), counts_(width * height, 0) {}

void TileMerger::Accumulate(const TileWindow& window, const std::vector<float>& probabilities) {
    if (probabilities.size() != static_cast<size_t>(window.width * window.height)) {
        throw std::invalid_argument("Tile probability size mismatch");
    }

    for (int y = 0; y < window.height; ++y) {
        for (int x = 0; x < window.width; ++x) {
            const int full_x = window.x + x;
            const int full_y = window.y + y;
            const int full_index = full_y * width_ + full_x;
            const int tile_index = y * window.width + x;
            sums_[full_index] += probabilities[tile_index];
            counts_[full_index] += 1;
        }
    }
}

std::vector<float> TileMerger::BuildProbability() const {
    std::vector<float> result(sums_.size(), 0.0f);
    for (size_t i = 0; i < result.size(); ++i) {
        result[i] = counts_[i] == 0 ? 0.0f : sums_[i] / static_cast<float>(counts_[i]);
    }
    return result;
}

std::vector<uint8_t> TileMerger::BuildMask(float threshold) const {
    auto probability = BuildProbability();
    std::vector<uint8_t> mask(probability.size(), 0);
    for (size_t i = 0; i < probability.size(); ++i) {
        mask[i] = probability[i] >= threshold ? 1 : 0;
    }
    return mask;
}

} // namespace gis_ai
```

`src/fusion/segmentation_pipeline.cpp`

```cpp
#include "fusion/segmentation_pipeline.h"

#include "fusion/tile_planner.h"
#include "fusion/tile_merger.h"
#include "io/raster_io.h"
#include "io/vector_io.h"
#include "ai/model_manager.h"
#include "ai/inference_engine.h"
#include "ai/preprocess.h"
#include "ai/postprocess.h"
#include "ai/mask_to_polygon.h"

#include <filesystem>
#include <fstream>

namespace gis_ai {

static std::string BuildBaseName(const std::string& input_path) {
    return std::filesystem::path(input_path).stem().string();
}

SegmentationResult RunSegmentation(const SegmentationOptions& options) {
    RasterIO raster_io;
    auto input = raster_io.Load(options.input_path);

    TilePlanner planner;
    auto windows = planner.Plan(input->width, input->height, options.tile_size, options.overlap);

    ModelManager manager;
    const std::string model_name = std::filesystem::path(options.model_path).stem().string();
    manager.LoadModel(options.model_path, model_name);
    InferenceEngine engine(manager);
    Preprocess preprocess;
    Postprocess postprocess;
    TileMerger merger(input->width, input->height);

    double total_inference_ms = 0.0;
    for (const auto& window : windows) {
        auto tile = RasterIO::ExtractWindow(*input, window);
        PreprocessConfig config;
        config.target_width = window.width;
        config.target_height = window.height;
        auto tensor = preprocess.RasterToTensor(*tile, config);
        auto shape = Preprocess::GetInputShape(config);
        auto infer_result = engine.Run(model_name, tensor, shape);
        total_inference_ms += infer_result.inference_time_ms;
        auto probability = postprocess.Sigmoid(infer_result.outputs[0]);
        merger.Accumulate(window, probability);
    }

    auto mask = merger.BuildMask(options.threshold);
    RasterData mask_raster = postprocess.MaskToRaster(mask, input->width, input->height, input->geotransform, input->projection);

    const std::string stem = BuildBaseName(options.input_path);
    std::filesystem::create_directories(options.output_path);
    const std::string mask_path = (std::filesystem::path(options.output_path) / (stem + "_mask.tif")).string();
    const std::string vector_path = (std::filesystem::path(options.output_path) / (stem + "_polygon.gpkg")).string();
    const std::string summary_path = (std::filesystem::path(options.output_path) / (stem + "_summary.json")).string();

    raster_io.Save(mask_raster, mask_path, RasterWriteOptions{options.compress});

    SegmentationResult result;
    result.input_path = options.input_path;
    result.output_mask_path = mask_path;
    result.output_summary_path = summary_path;
    result.tile_count = static_cast<int>(windows.size());
    result.inference_time_ms = total_inference_ms;
    result.success = true;

    try {
        if (options.vectorize) {
            MaskToPolygon mask_to_polygon;
            auto polygons = mask_to_polygon.Execute(mask, input->width, input->height, input->geotransform, 1);
            polygons->projection = input->projection;
            VectorIO vector_io;
            vector_io.Save(*polygons, vector_path);
            result.output_vector_path = vector_path;
        }
    } catch (const std::exception& ex) {
        result.partial_success = true;
        result.success = false;
        result.error_message = ex.what();
    }

    std::ofstream out(summary_path);
    out << "{\n"
        << "  \"input_path\": \"" << result.input_path << "\",\n"
```

```cpp
BatchSegmentationResult RunBatchSegmentation(const SegmentationOptions& options) {
    BatchSegmentationResult result;
    std::filesystem::create_directories(options.output_path);
    std::ofstream summary(std::filesystem::path(options.output_path) / "batch_summary.csv");
    summary << "input,status,tile_count,inference_time_ms,mask_path,vector_path,error\n";

    for (const auto& entry : std::filesystem::directory_iterator(options.input_path)) {
        if (!entry.is_regular_file()) {
            continue;
        }

        auto ext = entry.path().extension().string();
        std::transform(ext.begin(), ext.end(), ext.begin(), ::tolower);
        if (ext != ".tif" && ext != ".tiff") {
            continue;
        }

        SegmentationOptions single = options;
        single.input_path = entry.path().string();
        single.output_path = (std::filesystem::path(options.output_path) / entry.path().stem()).string();

        SegmentationResult item;
        try {
            item = RunSegmentation(single);
        } catch (const std::exception& ex) {
            item.input_path = single.input_path;
            item.error_message = ex.what();
        }

        const std::string status = item.success ? "success" : (item.partial_success ? "partial_success" : "failed");
        summary << item.input_path << "," << status << "," << item.tile_count << ","
                << item.inference_time_ms << "," << item.output_mask_path << ","
                << item.output_vector_path << "," << item.error_message << "\n";

        result.total += 1;
        result.succeeded += item.success ? 1 : 0;
        result.partial_succeeded += item.partial_success ? 1 : 0;
        result.failed += (!item.success && !item.partial_success) ? 1 : 0;
        result.items.push_back(std::move(item));
    }
    return result;
}
```

```cpp
std::vector<BatchResult> BatchProcessor::ProcessDirectory(const std::string& input_dir,

    SegmentationOptions options;
    options.input_path = input_dir;
    options.output_path = output_dir;

    }
    return results;
}
```

```cpp
#include "fusion/segmentation_pipeline.h"

#include <iostream>

int main(int argc, char** argv) {
    if (argc < 7) {
        std::cerr << "用法: gis_ai_segment --input <path> --output <dir> --model <model.onnx>\n";
        return 1;
    }

    gis_ai::SegmentationOptions options;
    for (int i = 1; i + 1 < argc; i += 2) {
        const std::string key = argv[i];
        const std::string value = argv[i + 1];
        if (key == "--input") options.input_path = value;
        else if (key == "--output") options.output_path = value;
        else if (key == "--model") options.model_path = value;
        else if (key == "--tile-size") options.tile_size = std::stoi(value);
        else if (key == "--overlap") options.overlap = std::stoi(value);
        else if (key == "--threshold") options.threshold = std::stof(value);
    }

    const bool is_directory = std::filesystem::is_directory(options.input_path);
    if (is_directory) {
        auto result = gis_ai::RunBatchSegmentation(options);
        return result.failed == 0 ? 0 : 2;
    }

    auto result = gis_ai::RunSegmentation(options);
    return (result.success || result.partial_success) ? 0 : 2;
}
```

```cpp
GIS_AI_API int GisAi_RasterSeg_Run(GisAiRasterSeg* seg, const char* input_tif,
                                   const char* output_tif, const char* output_vector) {
    if (!seg || !input_tif || !output_tif) { SetError(-2, "Null argument"); return -2; }
    try {
        gis_ai::SegmentationOptions options;
        options.input_path = input_tif;
        options.output_path = std::filesystem::path(output_tif).parent_path().string();
        options.model_path = seg->manager->GetModelPath(seg->model_name);
        options.vectorize = output_vector != nullptr;

        auto result = gis_ai::RunSegmentation(options);
        if (!result.success && !result.partial_success) {
            SetError(-3, result.error_message);
            return -3;
        }
        ClearError();
        return 0;
    } catch (const std::exception& e) {
        SetError(-3, e.what());
        return -3;
    }
}
```
